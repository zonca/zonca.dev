  </code>
 </li>
 <br/>
 <li>
  start ipython 0.11:
  <br/>
  <code>
   import PyTrilinos
   <br/>
   from IPython.kernel import client
   <br/>
   mec = client.MultiEngineClient()
   <br/>
   %load_ext parallelmagic
   <br/>
   mec.activate()
   <br/>
   px import PyTrilinos
   <br/>
   px comm=PyTrilinos.Epetra.PyComm()
   <br/>
   px print(comm.NumProc())
  </code>
